def _split_emails(raw: str):
    """Split email addresses by comma/semicolon/newline/whitespace."""
    if not raw:
        return []
    tmp = raw.replace(";", ",").replace("\n", ",").replace("\r", ",").replace("\t", ",")
    parts = [p.strip() for p in tmp.split(",") if p.strip()]
    return parts

def login_get_session(account: str, password: str) -> str:
    headers = {
        "accept": "application/json, text/plain, */*",
        "content-type": "application/x-www-form-urlencoded",
        "appid": "Hik-Connect-Portal",
        "clientsource": CLIENTSOURCE,
        "clienttype": CLIENTTYPE,
        "customno": CUSTOMNO,
        "featurecode": FEATURECODE,
        "user-agent": "Mozilla/5.0"
    }
    payload = {"checkSign": "false", "cuName": "d2Vi", "account": account, "password": password, "imageCode": ""}
    r = requests.post(HIK_LOGIN_URL, headers=headers, data=payload, timeout=20)
    r.raise_for_status()
    j = r.json()
    sessionid = (
        j.get("loginSession", {}).get("sessionId")
        or j.get("data", {}).get("token")
        or j.get("sessionid")
    )
    if not sessionid:
        raise ValueError(f"Session ID not in login response: {j}")
    return sessionid

def fetch_device_list(sessionid: str, offset: int = 0, limit: int = PAGE_SIZE) -> list:
    headers = {
        "accept": "application/json, text/plain, */*",
        "clientsource": CLIENTSOURCE,
        "clienttype": CLIENTTYPE,
        "customno": CUSTOMNO,
        "featurecode": FEATURECODE,
        "sessionid": sessionid,
        "user-agent": "Mozilla/5.0",
        "referer": "https://www.hik-connect.com/views/main/index.html",
    }
    params = {"limit": limit, "offset": offset}
    r = requests.get(HIK_DEVICE_LIST_URL, headers=headers, params=params, timeout=15)
    if r.status_code == 401:
        raise requests.HTTPError("401 – session expired")
    r.raise_for_status()
    json_resp = r.json()
    return json_resp.get("data", {}).get("data", []) or []

def fetch_all_devices(sessionid: str) -> dict:
    all_devices = {}
    offset = 0
    while True:
        devices = fetch_device_list(sessionid, offset)
        if not devices:
            break
        for dev in devices:
            all_devices[dev.get("subSerial")] = dev
        offset += PAGE_SIZE
        time.sleep(0.25)
    return all_devices

def generate_pdf(device_data: list, output_path: str):
    """Generate a colored PDF for the given device_data list."""
    # Lazy import ReportLab so the app can start without it
    try:
        from reportlab.lib.pagesizes import A4
        from reportlab.lib import colors
        from reportlab.lib.styles import getSampleStyleSheet, ParagraphStyle
        from reportlab.lib.enums import TA_CENTER, TA_LEFT
        from reportlab.platypus import SimpleDocTemplate, Table, TableStyle, Paragraph, Spacer, Image
    except Exception as e:
        raise RuntimeError(
            "ReportLab is required to generate PDFs. Install it with:\n"
            "    pip install reportlab\n"
            f"Details: {e}"
        )

    doc = SimpleDocTemplate(
        output_path,
        pagesize=A4,
        rightMargin=40,
        leftMargin=40,
        topMargin=40,
        bottomMargin=40
    )
    elements = []
    styles = getSampleStyleSheet()

    if os.path.isfile(LOGO_FILE):
        elements.append(Image(LOGO_FILE, width=100, height=50, hAlign="LEFT"))
        elements.append(Spacer(1, 10))

    header_style = ParagraphStyle(name="CenteredHeader", parent=styles["Heading1"], alignment=TA_CENTER)
    elements.append(Paragraph("CCTV Device Status Report", header_style))
    elements.append(Paragraph(
        f"Generated on: {datetime.now().strftime('%Y-%m-%d %H:%M:%S')} IST",
        ParagraphStyle(name="meta", fontSize=9, alignment=TA_LEFT)
    ))
    elements.append(Spacer(1, 12))

    table_data = [["S.No", "Branch Name", "Serial Number", "Status", "Remark"]]
    for i, d in enumerate(device_data, start=1):
        table_data.append([i, d.get("branch", ""), d.get("serial", ""), d.get("status", ""), d.get("remark", "-")])

    col_widths = [30, 150, 150, 70, 160]
    table = Table(table_data, colWidths=col_widths, repeatRows=1)
    tbl_style = TableStyle([
        ("BACKGROUND", (0, 0), (-1, 0), colors.HexColor("#1f6feb")),
        ("TEXTCOLOR", (0, 0), (-1, 0), colors.white),
        ("ALIGN", (0, 0), (-1, -1), "CENTER"),
        ("VALIGN", (0, 0), (-1, -1), "MIDDLE"),
        ("GRID", (0, 0), (-1, -1), 0.5, colors.HexColor("#c9d1d9")),
        ("FONTNAME", (0, 0), (-1, 0), "Helvetica-Bold"),
        ("FONTSIZE", (0, 0), (-1, -1), 10),
    ])

    for row_idx, row in enumerate(table_data[1:], start=1):
        status = str(row[3]).lower()
        if status == "online":
            bg = colors.HexColor("#DFF2E1"); txt = colors.HexColor("#1A7F2A")
        elif status == "offline":
            bg = colors.HexColor("#F8D7DA"); txt = colors.HexColor("#8B1A21")
        else:
            bg = colors.HexColor("#FFF3CD"); txt = colors.HexColor("#856404")
        tbl_style.add("BACKGROUND", (3, row_idx), (3, row_idx), bg)
        tbl_style.add("TEXTCOLOR", (3, row_idx), (3, row_idx), txt)

    table.setStyle(tbl_style)
    elements.append(table)
    doc.build(elements)

def send_email(subject, plain_text_body, html_body, to_email, cc_email, attachment_path):
    msg = MIMEMultipart("mixed")
    msg["From"] = SMTP_USER
    msg["To"] = to_email
    if cc_email:
        msg["Cc"] = cc_email
    msg["Subject"] = subject

    alt = MIMEMultipart("alternative")
    alt.attach(MIMEText(plain_text_body or "Please see the attached report.", "plain", "utf-8"))
    alt.attach(MIMEText(html_body or "<p>Please see the attached report.</p>", "html", "utf-8"))
    msg.attach(alt)

    if attachment_path and os.path.isfile(attachment_path):
        with open(attachment_path, "rb") as f:
            part = MIMEBase("application", "pdf")
            part.set_payload(f.read())
        encoders.encode_base64(part)
        part.add_header("Content-Disposition", "attachment", filename=os.path.basename(attachment_path))
        msg.attach(part)

    recipients = _split_emails(to_email)
    if cc_email:
        recipients += _split_emails(cc_email)

    context = ssl.create_default_context()
    with smtplib.SMTP_SSL(SMTP_SERVER, SMTP_PORT, context=context) as server:
        server.login(SMTP_USER, SMTP_PASSWORD)
        server.sendmail(SMTP_USER, recipients, msg.as_string())

    def _upload_excel_emails(self):
        """Open file dialog to select Excel file and extract email addresses."""
        file_path = filedialog.askopenfilename(
            title="Select Excel File with Email Addresses",
            filetypes=[("Excel files", "*.xlsx *.xls"), ("All files", "*.*")]
        )
        if not file_path:
            return
            
        try:
            # Read the Excel file
            df = pd.read_excel(file_path)
            
            # Find columns that might contain emails
            email_columns = []
            for col in df.columns:
                if any(x in str(col).lower() for x in ['email', 'mail', 'e-mail']):
                    email_columns.append(col)
            
            if not email_columns:
                # If no obvious email columns, try to find any string column with @ symbol
                for col in df.columns:
                    if df[col].dtype == 'object' and df[col].astype(str).str.contains('@').any():
                        email_columns.append(col)
                        break
            
            if not email_columns:
                self._alert_info("No Email Column", "Could not find any column containing email addresses.")
                return
                
            # Get unique emails from all identified columns
            emails = set()
            for col in email_columns:
                emails.update(df[col].dropna().astype(str).str.strip())
            
            # Filter valid emails
            valid_emails = [e for e in emails if '@' in e and '.' in e.split('@')[-1]]
            
            if not valid_emails:
                self._alert_info("No Valid Emails", "No valid email addresses found in the selected file.")
                return
                
            # Store the emails for later use
            self.uploaded_emails = ", ".join(valid_emails)
            self._alert_info("Success", f"Successfully loaded {len(valid_emails)} email addresses from {os.path.basename(file_path)}")
            
        except Exception as e:
            self._alert_error("Error", f"Failed to read Excel file: {str(e)}")

    def _safe_load_excel(self):
        if not os.path.isfile(EXCEL_FILE):
            self.df_excel = pd.DataFrame(columns=["State Name", "Branch Name", "Serial Number", "Remark"])
            self.device_rows = []
            self.serial_to_state = {}
            self._apply_filter()
            self._set_status(f"Excel not found: {EXCEL_FILE}. Starting with empty list.")
            return

        df = None
        for attempt in range(8):
            try:
                df = self._read_excel_with_fallback(EXCEL_FILE)
                break
            except PermissionError:
                self._set_status(f"Excel locked. Retrying ({attempt+1}/8)… Close '{EXCEL_FILE}'.")
                time.sleep(0.8)
            except Exception as e:
                self._set_status(f"Excel read error: {e}")
                break

        if df is None:
            self.df_excel = pd.DataFrame(columns=["State Name", "Branch Name", "Serial Number", "Remark"])
            self.device_rows = []
            self.serial_to_state = {}
            self._apply_filter()
            self._alert_info(
                "Excel Unavailable",
                f"Couldn't load '{EXCEL_FILE}'. The app started with empty list. "
                "Use 'Login + Full Status Refresh' or try again later."
            )
            return

        try:
            df.columns = [c.strip() for c in df.columns]
            if "Serial Number" not in df.columns:
                raise ValueError("Excel must have a 'Serial Number' column.")
            df["Serial Number"] = df["Serial Number"].astype(str).str.strip()
        except Exception as e:
            self.df_excel = pd.DataFrame(columns=["State Name", "Branch Name", "Serial Number", "Remark"])
            self.device_rows = []
            self.serial_to_state = {}
            self._apply_filter()
            self._alert_error(
                "Excel Format",
                f"Invalid Excel format: {e}\n"
                "Expected columns at least: Serial Number (and optionally State Name, Branch Name, Remark)."
            )
            return

        state_col = "State Name" if "State Name" in df.columns else ("State" if "State" in df.columns else None)
        if state_col:
            self.serial_to_state = {
                str(r["Serial Number"]).strip(): (str(r[state_col]).strip() if pd.notna(r[state_col]) else "-")
                for _, r in df.iterrows()
            }
        else:
            self.serial_to_state = {}

        self.df_excel = df
        self.device_rows = []
        for _, r in df.iterrows():
            serial = str(r["Serial Number"]).strip()
            self.device_rows.append({
                "state": self.serial_to_state.get(serial, "-"),
                "branch": r.get("Branch Name", "") if "Branch Name" in df.columns else "",
                "serial": serial,
                "status": "Not Found",
                "remark": (
                    r.get("Remark")
                    if "Remark" in df.columns and pd.notna(r.get("Remark", "-")) and str(r.get("Remark", "-")).strip()
                    else "-"
                ),
            })
        self._apply_filter()

    def _ensure_offline_table(self, conn):
        try:
            with conn.cursor() as cur:
                cur.execute(
                    """
                    CREATE TABLE IF NOT EXISTS offline_logs (
                        id SERIAL PRIMARY KEY,
                        branch_name TEXT NOT NULL,
                        offline_date DATE NOT NULL,
                        created_at TIMESTAMP WITHOUT TIME ZONE DEFAULT now(),
                        UNIQUE (branch_name, offline_date)
                    )
                    """
                )
            conn.commit()
        except Exception as e:
            self._set_status(f"DB table check failed: {e}")

    def _log_offline_branches_to_db(self):
        """Insert today's offline branches into DB (skip duplicates)."""
        offline_branches = sorted({
            r["branch"]
            for r in self.device_rows
            if str(r.get("status", "")).lower() in ("offline", "not found")
            and str(r.get("branch", "")).strip()
        })
        if not offline_branches:
            return
        conn = self._get_db_conn()
        if not conn:
            return
        try:
            self._ensure_offline_table(conn)
            today = date.today()
            with conn.cursor() as cur:
                for b in offline_branches:
                    cur.execute(
                        """
                        INSERT INTO offline_logs (branch_name, offline_date)
                        VALUES (%s, %s)
                        ON CONFLICT (branch_name, offline_date) DO NOTHING
                        """,
                        (b, today),
                    )
            conn.commit()
        except Exception as e:
            self._set_status(f"DB log insert failed: {e}")
        finally:
            try:
                conn.close()
            except Exception:
                pass

    def _update_offline_remarks_from_db(self):
        """
        For offline/not-found rows where remark is '-' or empty,
        look at offline_logs table and compute consecutive offline days streak.
        """
        branches = {
            r["branch"]
            for r in self.device_rows
            if str(r.get("status", "")).lower() in ("offline", "not found")
            and str(r.get("branch", "")).strip()
        }
        if not branches:
            return
        conn = self._get_db_conn()
        if not conn:
            return
        try:
            self._ensure_offline_table(conn)
            today = date.today()
            cache = {}
            with conn.cursor() as cur:
                for branch in branches:
                    cur.execute(
                        """
                        SELECT offline_date
                        FROM offline_logs
                        WHERE branch_name = %s AND offline_date <= %s
                        """,
                        (branch, today),
                    )
                    dates = [row[0] for row in cur.fetchall()]
                    cache[branch] = set(dates)

            for r in self.device_rows:
                if str(r.get("status", "")).lower() not in ("offline", "not found"):
                    continue
                # keep manual remarks if user already set something
                current_remark = str(r.get("remark", "")).strip()
                if current_remark and current_remark != "-":
                    continue
                branch = r.get("branch", "")
                if not branch:
                    continue
                date_set = cache.get(branch)
                if not date_set:
                    continue
                if today not in date_set:
                    continue
                # compute consecutive streak ending today
                streak = 1
                d = today - timedelta(days=1)
                while d in date_set:
                    streak += 1
                    d -= timedelta(days=1)
                if streak <= 0:
                    continue
                if streak == 1:
                    text = "CCTV has been offline since today."
                else:
                    text = f"CCTV has been offline from last {streak} days."
                r["remark"] = text
        except Exception as e:
            self._set_status(f"DB remark update failed: {e}")
        finally:
            try:
                conn.close()
            except Exception:
                pass

    # ---------- Core status update ----------
    def _update_statuses_from_latest(self, latest_map: dict):
        serial_to_status = {
            s: ("Online" if d.get("status", 0) == 1 else "Offline" if d.get("status", 0) == 2 else "Not Found")
            for s, d in latest_map.items()
        }
        changed = 0
        for row in self.device_rows:
            new_status = serial_to_status.get(row["serial"], "Not Found")
            if new_status != row["status"]:
                row["status"] = new_status
                changed += 1
            if new_status == "Online":
                row["remark"] = "-"
        # Log offline branches into DB and then auto-update remarks based on streak
        try:
            self._log_offline_branches_to_db()
            self._update_offline_remarks_from_db()
        except Exception:
            pass
        self._apply_filter()
        return changed

    # ---------- Auto refresh ----------
    def _auto_refresh_tick(self):
        def worker():
            try:
                self._set_status("Auto refresh: logging in / fetching…")
                if self.sessionid is None:
                    self.sessionid = login_get_session(ACCOUNT, PASSWORD)
                try:
                    latest_map = fetch_all_devices(self.sessionid)
                except requests.HTTPError:
                    self.sessionid = login_get_session(ACCOUNT, PASSWORD)
                    latest_map = fetch_all_devices(self.sessionid)
                changed = self._update_statuses_from_latest(latest_map)
                ts = datetime.now().strftime('%H:%M:%S')
                self._set_status(f"Auto-refreshed {changed} rows at {ts} (every 5 min)")
                self.lbl_last_status.config(text=f"Last refresh: {ts}")
            except Exception as e:
                self._set_status(f"Auto-refresh error: {e}")

        threading.Thread(target=worker, daemon=True).start()
        self.after(AUTO_REFRESH_MS, self._auto_refresh_tick)

                self.sessionid = login_get_session(ACCOUNT, PASSWORD)
                latest_map = fetch_all_devices(self.sessionid)

    # ---------- Blocking refresh helper (for PDF/Email actions) ----------
    def _run_refresh_blocking(self):
        done = tk.BooleanVar(value=False)
        error = {"msg": None}

        win = tk.Toplevel(self)
        win.title("Refreshing…")
        win.transient(self)
        win.grab_set()
        win.resizable(False, False)
        ttk.Label(win, text="Refreshing latest status from Hik-Connect…", padding=14)\
            .grid(row=0, column=0, padx=12, pady=(12, 6))
        pb = ttk.Progressbar(win, mode="indeterminate", length=260)
        pb.grid(row=1, column=0, padx=12, pady=(0, 12))
        pb.start(12)

        def worker():
            try:
                if self.sessionid is None:
                    self.sessionid = login_get_session(ACCOUNT, PASSWORD)
                try:
                    latest_map = fetch_all_devices(self.sessionid)
                except requests.HTTPError:
                    self.sessionid = login_get_session(ACCOUNT, PASSWORD)
                    latest_map = fetch_all_devices(self.sessionid)
                self._update_statuses_from_latest(latest_map)
            except Exception as e:
                error["msg"] = str(e)
            finally:
                self.after(0, lambda: done.set(True))

        threading.Thread(target=worker, daemon=True).start()
        self.wait_variable(done)
        pb.stop()
        win.destroy()
        return (error["msg"] is None, error["msg"])

    def save_to_excel(self):
        try:
            if self.df_excel is None or "Serial Number" not in getattr(self.df_excel, "columns", []):
                self._alert_info("Save", "No Excel model to save yet.")
                return
            df = self.df_excel.copy()

            df["Remark"] = df["Serial Number"].astype(str).str.strip().map(serial_to_remark).fillna(df.get("Remark"))
            for attempt in range(10):
                try:
                    df.to_excel(EXCEL_FILE, index=False)
                    break
                except PermissionError:
                    self._set_status(f"Excel is locked for writing. Retrying save ({attempt+1}/10)…")
                    time.sleep(1)
            else:
                self._alert_error(
                    "Save Failed",
                    f"Could not save '{EXCEL_FILE}' because it is in use.\nClose it in Excel/OneDrive and try again."
                )
                return
            self.df_excel = df
            self._alert_info("Saved", f"Remarks saved to {EXCEL_FILE}")
        except Exception as e:
            self._alert_error("Error", f"Failed to save Excel: {e}")

            self._alert_info("Not Allowed", "Remarks can be edited only for OFFLINE or NOT FOUND devices.")

            f"State: {state}\nBranch: {branch}\nSerial: {serial}\nStatus: {status}\n\nEnter remark:",

        if new_remark is None:
            return
        new_remark = new_remark.strip() if new_remark.strip() else "-"
        for r in self.device_rows:
            if r["serial"] == serial and r["branch"] == branch:
                r["remark"] = new_remark
                r["state"] = self.serial_to_state.get(serial, r.get("state", "-")) or "-"
                break
        self._apply_filter()

    def _get_filtered_rows_for_pdf(self):
        if not self.device_rows:
            self._alert_info("PDF", "No data.")
            return None, None

        include_online = self.var_pdf_online.get()
        include_offline = self.var_pdf_offline.get()
        if not include_online and not include_offline:
            self._alert_info("PDF", "Please tick at least one option: Online and/or Offline.")
            return None, None

        def is_offish(st: str) -> bool:
            return (st or "").lower() in ("offline", "not found")

        filtered = []
        for d in self.device_rows:
            s = (d.get("status", "")).lower()
            if include_online and s == "online":
                filtered.append(d)
            elif include_offline and is_offish(s):
                filtered.append(d)

        if not filtered:
            self._alert_info("PDF", "No matching rows for the selected filter.")
            return None, None

        suffix = "ALL"
        if include_online and not include_offline:
            suffix = "ONLINE"
        if include_offline and not include_online:
            suffix = "OFFLINE"
        return filtered, suffix

    # ---------- Filtering & painting ----------
    def _apply_filter(self):
        for r in self.device_rows:
            r["state"] = self.serial_to_state.get(str(r.get("serial", "")).strip(), r.get("state", "-")) or "-"

        q = self.var_search.get().strip().lower()
        data = self.device_rows
        if q and not q.startswith("type to filter"):
            data = [
                r for r in self.device_rows
                if q in str(r.get("state", "-")).lower()
                or q in str(r["branch"]).lower()
                or q in str(r["serial"]).lower()
            ]
        self._refresh_split_tables(data)

    def _refresh_split_tables(self, data):
        online = [r for r in data if str(r["status"]).lower() == "online"]
        offish = [r for r in data if str(r["status"]).lower() in ("offline", "not found")]

        for tv in (self.left_tree, self.right_tree):
            for item in tv.get_children():
                tv.delete(item)

        for idx, r in enumerate(online):
            tags = ["online"]
            if idx % 2 == 1:
                tags.append("odd")
            self.left_tree.insert(
                "",
                tk.END,
                values=(r.get("state", "-"), r["branch"], r["serial"], r["status"], r["remark"]),
                tags=tuple(tags)
            )

        for idx, r in enumerate(offish):
            t = "offline" if str(r["status"]).lower() == "offline" else "notfound"
            tags = [t]
            if idx % 2 == 1:
                tags.append("odd")
            self.right_tree.insert(
                "",
                tk.END,
                values=(r.get("state", "-"), r["branch"], r["serial"], r["status"], r["remark"]),
                tags=tuple(tags)
            )

        self.lbl_left.config(text=f"Online ({len(online)})")
        self.lbl_right.config(text=f"Offline / Not Found ({len(offish)})")
        self._set_status(f"Online: {len(online)} | Offline/Not Found: {len(offish)}")

    # ---------- Sorting ----------
    def _sort_treeview(self, tree: ttk.Treeview, col: str, reverse: bool):
        rows = [(tree.set(k, col), k) for k in tree.get_children("")]

        def to_key(v):
            s = str(v).strip()
            if col in ("serial",):
                return (len(s), s)
            return s.lower()

        rows.sort(key=lambda x: to_key(x[0]), reverse=reverse)
        for idx, (_, k) in enumerate(rows):
            tree.move(k, "", idx)
        tree.heading(col, command=lambda: self._sort_treeview(tree, col, not reverse))
